371def main():
372    conn = sqlite3.connect(DB_PATH)
373    cur = conn.cursor()
374
375    print("Seeding dictionary terms...")
376    inserted = 0
377    for t in TERMS:
378        cur.execute("""
379            INSERT OR IGNORE INTO dictionary_terms
380                (slug, label, category, definition_short, definition_long,
381                 source_basis, review_status, needs_review)
382            VALUES (?, ?, ?, ?, ?, ?, 'DRAFT', 1)
383        """, (t["slug"], t["label"], t["category"],
384              t["definition_short"], t["definition_long"],
385              t.get("source_basis", "")))
386        inserted += cur.rowcount
387    conn.commit()
388    print(f"  Inserted {inserted} terms ({len(TERMS)} total defined)")
389
390    print("Creating term links...")
391    linked = 0
392    for slug_a, slug_b, link_type in LINKS:
393        cur.execute("SELECT id FROM dictionary_terms WHERE slug=?", (slug_a,))
394        a = cur.fetchone()
395        cur.execute("SELECT id FROM dictionary_terms WHERE slug=?", (slug_b,))
396        b = cur.fetchone()
397        if a and b:
398            cur.execute("""
399                INSERT OR IGNORE INTO dictionary_term_links
400                    (term_id, linked_term_id, link_type)
401                VALUES (?, ?, ?)
402            """, (a[0], b[0], link_type))
403            linked += cur.rowcount
404            # Bidirectional for RELATED
405            if link_type == "RELATED":
406                cur.execute("""
407                    INSERT OR IGNORE INTO dictionary_term_links
408                        (term_id, linked_term_id, link_type)
409                    VALUES (?, ?, ?)
410                """, (b[0], a[0], link_type))
411                linked += cur.rowcount
412    conn.commit()
413    print(f"  Created {linked} links")
414
415    # Summary
416    print("\nDictionary by category:")
417    cur.execute("""
418        SELECT category, COUNT(*) FROM dictionary_terms
419        GROUP BY category ORDER BY COUNT(*) DESC
420    """)
421    for row in cur.fetchall():
422        print(f"  {row[0]}: {row[1]}")
423
424    conn.close()
425    print("\nDone.")
