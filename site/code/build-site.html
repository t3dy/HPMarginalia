31def slugify(text):
32    s = text.lower().strip()
33    s = re.sub(r'[^\w\s-]', '', s)
34    s = re.sub(r'[\s_]+', '-', s)
35    s = re.sub(r'-+', '-', s)
36    return s.strip('-')

39def nav_html(active='', prefix=''):

148def export_data_json(conn):
149    """Export marginalia gallery data with confidence flags."""
150    cur = conn.cursor()
151    cur.execute("""
152        SELECT
153            r.id, r.thesis_page, r.signature_ref, m.shelfmark,
154            m.institution, m.city, r.context_text, r.marginal_text,
155            r.chapter_num, i.filename, i.relative_path,
156            i.folio_number, i.side, mat.confidence,
157            sm.quire, sm.leaf_in_quire,
158            mat.needs_review as match_needs_review,
159            h.hand_label, h.attribution, h.is_alchemist
160        FROM matches mat
161        JOIN dissertation_refs r ON mat.ref_id = r.id
162        JOIN images i ON mat.image_id = i.id
163        JOIN manuscripts m ON i.manuscript_id = m.id
164        LEFT JOIN signature_map sm ON LOWER(r.signature_ref) = LOWER(sm.signature)
165        LEFT JOIN annotator_hands h ON r.hand_id = h.id
166        WHERE i.page_type = 'PAGE'
167        GROUP BY r.signature_ref, i.filename
168        ORDER BY COALESCE(sm.folio_number, 999), r.thesis_page
169    """)
170
171    entries = []
172    sigs = set()
173    for row in cur.fetchall():
174        entry = {
175            'ref_id': row[0], 'thesis_page': row[1],
176            'signature': row[2], 'manuscript': row[3],
177            'institution': row[4], 'city': row[5],
178            'context': (row[6] or '')[:600],
179            'marginal_text': row[7], 'chapter': row[8],
180            'image_file': row[9],
181            'image_path': row[10],
182            'folio_number': row[11], 'side': row[12],
183            'confidence': row[13] or 'PROVISIONAL',
184            'quire': row[14], 'leaf_in_quire': row[15],
185            'needs_review': bool(row[16]),
186            'hand_label': row[17], 'hand_attribution': row[18],
187            'is_alchemist': bool(row[19]) if row[19] is not None else False,
188        }
189        entries.append(entry)
190        sigs.add(row[2])
191
192    data = {
193        'entries': entries,
194        'stats': {
195            'total_references': len(entries),
196            'unique_signatures': len(sigs),
197            'high_confidence_matches': sum(1 for e in entries if e['confidence'] == 'HIGH'),
198            'low_confidence_matches': sum(1 for e in entries if e['confidence'] == 'LOW'),
199            'needs_review': sum(1 for e in entries if e['needs_review']),
200        },
201        'provenance': {
202            'source': 'hp.db v2',
203            'note': 'BL C.60.o.12 matches are LOW confidence (1545 edition, unverified photo-folio mapping)',
204        },
205    }
206
207    out = SITE_DIR / 'data.json'
208    with open(out, 'w', encoding='utf-8') as f:
209        json.dump(data, f, indent=2, ensure_ascii=False)
210    print(f"  data.json: {len(entries)} entries")

217def build_scholars_pages(conn):
218    """Generate scholars.html and scholar/*.html from DB + summaries.json."""
219    # Load summaries for detailed content
220    summaries = []
221    if SUMMARIES_PATH.exists():
222        with open(SUMMARIES_PATH, encoding='utf-8') as f:
223            summaries = json.load(f)
224
225    # Group by author
226    by_author = {}
227    for s in summaries:
228        author = s.get('author', 'Unknown')
229        by_author.setdefault(author, []).append(s)
230
231    cur = conn.cursor()
232    # Get review status from DB
233    cur.execute("SELECT name, needs_review, source_method FROM scholars")
234    scholar_review = {}
235    for row in cur.fetchall():
236        scholar_review[row[0]] = {'needs_review': row[1], 'source_method': row[2]}
237
238    # Build scholars index
239    scholar_cards = []
240    scholar_dir = SITE_DIR / 'scholar'
241    scholar_dir.mkdir(exist_ok=True)
242
243    for author in sorted(by_author.keys()):
244        papers = by_author[author]
245        slug = slugify(author)
246        review = scholar_review.get(author, {'needs_review': True, 'source_method': 'LLM_ASSISTED'})
247
248        # Collect all topics
249        topics = set()
250        for p in papers:
251            tc = p.get('topic_cluster', '')
252            if tc:
253                for t in tc.split(','):
254                    topics.add(t.strip())
255
256        badges = topic_badges_html(','.join(topics))
257        review_html = review_badge_html(review.get('needs_review'), review.get('source_method'))
258
259        # Paper cards for index
260        paper_cards_html = ''
261        for p in papers:
262            paper_slug = slugify(p.get('title', ''))
263            summary_preview = (p.get('summary', '') or '')[:250]
264            if len(p.get('summary', '')) > 250:
265                summary_preview += '...'
266            paper_cards_html += f"""
267                <div class="paper-card">
268                    <h4><a href="{slug}.html">{escape(p.get('title', ''))}</a></h4>
269                    <div class="paper-meta">{escape(p.get('journal', ''))} ({p.get('year', '?')})</div>
270                    <div class="paper-summary">{escape(summary_preview)}</div>
271                </div>"""
272
273        scholar_cards.append(f"""
274        <div class="scholar-card">
275            <h3><a href="{slug}.html">{escape(author)}</a> {review_html}</h3>
276            <div class="scholar-meta">{len(papers)} paper{'s' if len(papers) != 1 else ''} {badges}</div>
277            <div class="scholar-papers">{paper_cards_html}</div>
278        </div>""")
279
280        # Build individual scholar page
281        papers_detail = ''
282        for p in papers:
283            tc = p.get('topic_cluster', '')
284            papers_detail += f"""
285            <div class="paper-detail">
286                <h3>{escape(p.get('title', ''))}</h3>
287                <div class="paper-meta">{escape(p.get('journal', ''))} ({p.get('year', '?')})
288                    {topic_badges_html(tc)}</div>
289                <div class="paper-summary-full"><p>{escape(p.get('summary', ''))}</p></div>
290            </div>"""
291
292        detail_body = f"""
293        <div class="scholar-detail">
294            <h2>{escape(author)} {review_html}</h2>
295            <p><a href="../scholars.html">&larr; All Scholars</a></p>
296            {papers_detail}
297        </div>"""
298
299        detail_page = page_shell(author, detail_body, active_nav='scholars', depth=1)
300        (scholar_dir / f'{slug}.html').write_text(detail_page, encoding='utf-8')
301
302    # Build index page
303    index_body = f"""
304        <div class="scholars-grid">
305            <div class="intro" style="margin-bottom:2rem">
306                <p>{len(by_author)} scholars, {len(summaries)} works documented.
307                Data sourced from SQLite database with provenance tracking.</p>
308            </div>
309            {''.join(scholar_cards)}
310        </div>"""
311
312    index_page = page_shell('Scholars', index_body, active_nav='scholars')
313    (SITE_DIR / 'scholars.html').write_text(index_page, encoding='utf-8')
314    print(f"  scholars.html + {len(by_author)} scholar pages")

906def markdown_to_html(md_text):
907    """Minimal markdown-to-HTML conversion for document display."""
908    import re
909    lines = md_text.split('\n')
910    html_lines = []
911    in_code = False
912    in_list = False
913    in_table = False
914    table_header_done = False
915
916    for line in lines:
917        # Code blocks
918        if line.strip().startswith('```'):
919            if in_code:
920                html_lines.append('</code></pre>')
921                in_code = False
922            else:
923                lang = line.strip()[3:]
924                html_lines.append(f'<pre><code class="lang-{lang}">')
925                in_code = True
926            continue
927        if in_code:
928            html_lines.append(escape(line))
929            continue
930
931        # Close list if needed
932        if in_list and not line.strip().startswith(('-', '*', '1.', '2.', '3.', '4.', '5.', '6.', '7.', '8.', '9.')):
933            html_lines.append('</ul>')
934            in_list = False
935
936        # Tables
937        if '|' in line and line.strip().startswith('|'):
938            cells = [c.strip() for c in line.strip().split('|')[1:-1]]
939            if all(set(c) <= set('- :') for c in cells):
940                table_header_done = True
941                continue
942            if not in_table:
943                html_lines.append('<table class="doc-table">')
944                in_table = True
945            tag = 'th' if not table_header_done else 'td'
946            row = ''.join(f'<{tag}>{escape(c)}</{tag}>' for c in cells)
947            html_lines.append(f'<tr>{row}</tr>')
948            continue
949        elif in_table:
950            html_lines.append('</table>')
951            in_table = False
952            table_header_done = False
953
954        stripped = line.strip()
955
956        # Headings
957        if stripped.startswith('### '):
958            html_lines.append(f'<h4>{escape(stripped[4:])}</h4>')
959        elif stripped.startswith('## '):
960            html_lines.append(f'<h3>{escape(stripped[3:])}</h3>')
961        elif stripped.startswith('# '):
962            html_lines.append(f'<h2>{escape(stripped[2:])}</h2>')
963        elif stripped.startswith('---'):
964            html_lines.append('<hr>')
965        elif stripped.startswith(('- ', '* ')):
966            if not in_list:
967                html_lines.append('<ul>')
968                in_list = True
969            html_lines.append(f'<li>{escape(stripped[2:])}</li>')
970        elif stripped.startswith('>'):
971            html_lines.append(f'<blockquote>{escape(stripped[1:].strip())}</blockquote>')
972        elif stripped == '':
973            html_lines.append('')
974        else:
975            # Apply inline formatting
976            text = escape(stripped)
977            # Bold
978            text = re.sub(r'\*\*(.+?)\*\*', r'<strong>\1</strong>', text)
979            # Italic
980            text = re.sub(r'\*(.+?)\*', r'<em>\1</em>', text)
981            # Inline code
982            text = re.sub(r'`(.+?)`', r'<code>\1</code>', text)
983            html_lines.append(f'<p>{text}</p>')
984
985    if in_list:
986        html_lines.append('</ul>')
987    if in_table:
988        html_lines.append('</table>')
989    if in_code:
990        html_lines.append('</code></pre>')
991
992    return '\n'.join(html_lines)
