31def slugify(text):
32    s = text.lower().strip()
33    s = re.sub(r'[^\w\s-]', '', s)
34    s = re.sub(r'[\s_]+', '-', s)
35    s = re.sub(r'-+', '-', s)
36    return s.strip('-')

151def export_data_json(conn):
152    """Export marginalia gallery data with confidence flags."""
153    cur = conn.cursor()
154    cur.execute("""
155        SELECT
156            r.id, r.thesis_page, r.signature_ref, m.shelfmark,
157            m.institution, m.city, r.context_text, r.marginal_text,
158            r.chapter_num, i.filename, i.relative_path,
159            i.folio_number, i.side, mat.confidence,
160            sm.quire, sm.leaf_in_quire,
161            mat.needs_review as match_needs_review,
162            h.hand_label, h.attribution, h.is_alchemist
163        FROM matches mat
164        JOIN dissertation_refs r ON mat.ref_id = r.id
165        JOIN images i ON mat.image_id = i.id
166        JOIN manuscripts m ON i.manuscript_id = m.id
167        LEFT JOIN signature_map sm ON LOWER(r.signature_ref) = LOWER(sm.signature)
168        LEFT JOIN annotator_hands h ON r.hand_id = h.id
169        WHERE i.page_type = 'PAGE'
170        GROUP BY r.signature_ref, i.filename
171        ORDER BY COALESCE(sm.folio_number, 999), r.thesis_page
172    """)
173
174    # Load folio descriptions for alchemist annotations
175    folio_descs = {}
176    try:
177        cur2 = conn.cursor()
178        cur2.execute("""
179            SELECT signature_ref, manuscript_shelfmark, hand_label,
180                   title, description, alchemical_element, alchemical_process,
181                   alchemical_framework, russell_page_ref
182            FROM folio_descriptions
183        """)
184        for fd in cur2.fetchall():
185            key = (fd[0], fd[1])
186            folio_descs[key] = {
187                'desc_title': fd[3], 'desc_text': fd[4],
188                'alch_element': fd[5], 'alch_process': fd[6],
189                'alch_framework': fd[7], 'russell_pages': fd[8],
190            }
191    except:
192        pass  # Table may not exist yet
193
194    entries = []
195    sigs = set()
196    for row in cur.fetchall():
197        sig = row[2]
198        ms = row[3]
199        entry = {
200            'ref_id': row[0], 'thesis_page': row[1],
201            'signature': sig, 'manuscript': ms,
202            'institution': row[4], 'city': row[5],
203            'context': (row[6] or '')[:600],
204            'marginal_text': row[7], 'chapter': row[8],
205            'image_file': row[9],
206            'image_path': row[10],
207            'folio_number': row[11], 'side': row[12],
208            'confidence': row[13] or 'PROVISIONAL',
209            'quire': row[14], 'leaf_in_quire': row[15],
210            'needs_review': bool(row[16]),
211            'hand_label': row[17], 'hand_attribution': row[18],
212            'is_alchemist': bool(row[19]) if row[19] is not None else False,
213        }
214        # Add folio description if available
215        fd = folio_descs.get((sig, ms)) or folio_descs.get((sig, None))
216        if fd:
217            entry['desc_title'] = fd['desc_title']
218            entry['desc_text'] = fd['desc_text']
219            entry['alch_element'] = fd['alch_element']
220        entries.append(entry)
221        sigs.add(sig)
222
223    data = {
224        'entries': entries,
225        'stats': {
226            'total_references': len(entries),
227            'unique_signatures': len(sigs),
228            'high_confidence_matches': sum(1 for e in entries if e['confidence'] == 'HIGH'),
229            'low_confidence_matches': sum(1 for e in entries if e['confidence'] == 'LOW'),
230            'needs_review': sum(1 for e in entries if e['needs_review']),
231        },
232        'provenance': {
233            'source': 'hp.db v2',
234            'note': 'BL C.60.o.12 matches are LOW confidence (1545 edition, unverified photo-folio mapping)',
235        },
236    }
237
238    out = SITE_DIR / 'data.json'
239    with open(out, 'w', encoding='utf-8') as f:
240        json.dump(data, f, indent=2, ensure_ascii=False)
241    print(f"  data.json: {len(entries)} entries")

248def build_scholars_pages(conn):
249    """Generate scholars.html and scholar/*.html from DB + summaries.json."""
250    # Load summaries for detailed content
251    summaries = []
252    if SUMMARIES_PATH.exists():

254            summaries = json.load(f)
255

257    by_author = {}
258    for s in summaries:
259        author = s.get('author', 'Unknown')
260        by_author.setdefault(author, []).append(s)
261
262    cur = conn.cursor()
263    # Get review status from DB
264    cur.execute("SELECT name, needs_review, source_method FROM scholars")
265    scholar_review = {}
266    for row in cur.fetchall():
267        scholar_review[row[0]] = {'needs_review': row[1], 'source_method': row[2]}
268
269    # Build scholars index
270    scholar_cards = []
271    scholar_dir = SITE_DIR / 'scholar'
272    scholar_dir.mkdir(exist_ok=True)
273
274    for author in sorted(by_author.keys()):
275        papers = by_author[author]
276        slug = slugify(author)
277        review = scholar_review.get(author, {'needs_review': True, 'source_method': 'LLM_ASSISTED'})
278
279        # Collect all topics
280        topics = set()
281        for p in papers:
282            tc = p.get('topic_cluster', '')
283            if tc:
284                for t in tc.split(','):
285                    topics.add(t.strip())
286
287        badges = topic_badges_html(','.join(topics))
288        review_html = review_badge_html(review.get('needs_review'), review.get('source_method'))
289
290        # Paper cards for index
291        paper_cards_html = ''
292        for p in papers:
293            paper_slug = slugify(p.get('title', ''))
294            summary_preview = (p.get('summary', '') or '')[:250]
295            if len(p.get('summary', '')) > 250:
296                summary_preview += '...'
297            paper_cards_html += f"""
298                <div class="paper-card">
299                    <h4><a href="{slug}.html">{escape(p.get('title', ''))}</a></h4>
300                    <div class="paper-meta">{escape(p.get('journal', ''))} ({p.get('year', '?')})</div>
301                    <div class="paper-summary">{escape(summary_preview)}</div>
302                </div>"""
303
304        scholar_cards.append(f"""
305        <div class="scholar-card">
306            <h3><a href="{slug}.html">{escape(author)}</a> {review_html}</h3>
307            <div class="scholar-meta">{len(papers)} paper{'s' if len(papers) != 1 else ''} {badges}</div>
308            <div class="scholar-papers">{paper_cards_html}</div>
309        </div>""")
310
311        # Build individual scholar page
312        papers_detail = ''
313        for p in papers:
314            tc = p.get('topic_cluster', '')
315            papers_detail += f"""
316            <div class="paper-detail">
317                <h3>{escape(p.get('title', ''))}</h3>
318                <div class="paper-meta">{escape(p.get('journal', ''))} ({p.get('year', '?')})
319                    {topic_badges_html(tc)}</div>
320                <div class="paper-summary-full"><p>{escape(p.get('summary', ''))}</p></div>
321            </div>"""
322
323        detail_body = f"""
324        <div class="scholar-detail">
325            <h2>{escape(author)} {review_html}</h2>
326            <p><a href="../scholars.html">&larr; All Scholars</a></p>
327            {papers_detail}
328        </div>"""
329
330        detail_page = page_shell(author, detail_body, active_nav='scholars', depth=1)
331        (scholar_dir / f'{slug}.html').write_text(detail_page, encoding='utf-8')
332
333    # Build index page
334    index_body = f"""
335        <div class="scholars-grid">
336            <div class="intro" style="margin-bottom:2rem">
337                <h2>Scholars of the <em>Hypnerotomachia</em></h2>
338                <p>The scholarship on the <em>Hypnerotomachia Poliphili</em> is wide-ranging
339                and interdisciplinary. It spans bibliography, book history, architecture,
340                garden studies, allegory, philology, reception history, emblematic reading,
341                and the history of interpretation. This section organizes that scholarly
342                landscape by person: each profile brings together the works in the collection
343                associated with a given scholar and situates them within the larger conversation.</p>
344                <p>These pages are meant to make the field legible at a glance.
345                {len(by_author)} scholars, {len(summaries)} article and monograph summaries.
346                Where content is LLM-assisted or otherwise provisional, that status
347                remains visible.</p>
348            </div>
349            {''.join(scholar_cards)}
350        </div>"""
351
352    index_page = page_shell('Scholars', index_body, active_nav='scholars')
353    (SITE_DIR / 'scholars.html').write_text(index_page, encoding='utf-8')
354    print(f"  scholars.html + {len(by_author)} scholar pages")

1128def markdown_to_html(md_text):
1129    """Minimal markdown-to-HTML conversion for document display."""
1130    import re
1131    lines = md_text.split('\n')
1132    html_lines = []
1133    in_code = False
1134    in_list = False
1135    in_table = False
1136    table_header_done = False
1137
1138    for line in lines:
1139        # Code blocks
1140        if line.strip().startswith('```'):
1141            if in_code:
1142                html_lines.append('</code></pre>')
1143                in_code = False
1144            else:
1145                lang = line.strip()[3:]
1146                html_lines.append(f'<pre><code class="lang-{lang}">')
1147                in_code = True
1148            continue
1149        if in_code:
1150            html_lines.append(escape(line))
1151            continue
1152
1153        # Close list if needed
1154        if in_list and not line.strip().startswith(('-', '*', '1.', '2.', '3.', '4.', '5.', '6.', '7.', '8.', '9.')):
1155            html_lines.append('</ul>')
1156            in_list = False
1157
1158        # Tables
1159        if '|' in line and line.strip().startswith('|'):
1160            cells = [c.strip() for c in line.strip().split('|')[1:-1]]
1161            if all(set(c) <= set('- :') for c in cells):
1162                table_header_done = True
1163                continue
1164            if not in_table:
1165                html_lines.append('<table class="doc-table">')
1166                in_table = True
1167            tag = 'th' if not table_header_done else 'td'
1168            row = ''.join(f'<{tag}>{escape(c)}</{tag}>' for c in cells)
1169            html_lines.append(f'<tr>{row}</tr>')
1170            continue
1171        elif in_table:
1172            html_lines.append('</table>')
1173            in_table = False
1174            table_header_done = False
1175
1176        stripped = line.strip()
1177
1178        # Headings
1179        if stripped.startswith('### '):
1180            html_lines.append(f'<h4>{escape(stripped[4:])}</h4>')
1181        elif stripped.startswith('## '):
1182            html_lines.append(f'<h3>{escape(stripped[3:])}</h3>')
1183        elif stripped.startswith('# '):
1184            html_lines.append(f'<h2>{escape(stripped[2:])}</h2>')
1185        elif stripped.startswith('---'):
1186            html_lines.append('<hr>')
1187        elif stripped.startswith(('- ', '* ')):
1188            if not in_list:
1189                html_lines.append('<ul>')
1190                in_list = True
1191            html_lines.append(f'<li>{escape(stripped[2:])}</li>')
1192        elif stripped.startswith('>'):
1193            html_lines.append(f'<blockquote>{escape(stripped[1:].strip())}</blockquote>')
1194        elif stripped == '':
1195            html_lines.append('')
1196        else:
1197            # Apply inline formatting
1198            text = escape(stripped)
1199            # Bold
1200            text = re.sub(r'\*\*(.+?)\*\*', r'<strong>\1</strong>', text)
1201            # Italic
1202            text = re.sub(r'\*(.+?)\*', r'<em>\1</em>', text)
1203            # Inline code
1204            text = re.sub(r'`(.+?)`', r'<code>\1</code>', text)
1205            html_lines.append(f'<p>{text}</p>')
1206
1207    if in_list:
1208        html_lines.append('</ul>')
1209    if in_table:
1210        html_lines.append('</table>')
1211    if in_code:
1212        html_lines.append('</code></pre>')
1213
1214    return '\n'.join(html_lines)
