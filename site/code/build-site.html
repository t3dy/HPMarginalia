31def slugify(text):
32    s = text.lower().strip()
33    s = re.sub(r'[^\w\s-]', '', s)
34    s = re.sub(r'[\s_]+', '-', s)
35    s = re.sub(r'-+', '-', s)
36    return s.strip('-')

155def export_data_json(conn):
156    """Export marginalia gallery data with confidence flags."""
157    cur = conn.cursor()
158    cur.execute("""
159        SELECT
160            r.id, r.thesis_page, r.signature_ref, m.shelfmark,
161            m.institution, m.city, r.context_text, r.marginal_text,
162            r.chapter_num, i.filename, i.relative_path,
163            i.folio_number, i.side, mat.confidence,
164            sm.quire, sm.leaf_in_quire,
165            mat.needs_review as match_needs_review,
166            h.hand_label, h.attribution, h.is_alchemist
167        FROM matches mat
168        JOIN dissertation_refs r ON mat.ref_id = r.id
169        JOIN images i ON mat.image_id = i.id
170        JOIN manuscripts m ON i.manuscript_id = m.id
171        LEFT JOIN signature_map sm ON LOWER(r.signature_ref) = LOWER(sm.signature)
172        LEFT JOIN annotator_hands h ON r.hand_id = h.id
173        WHERE i.page_type = 'PAGE'
174        GROUP BY r.signature_ref, i.filename
175        ORDER BY COALESCE(sm.folio_number, 999), r.thesis_page
176    """)
177
178    # Load folio descriptions for alchemist annotations
179    folio_descs = {}
180    try:
181        cur2 = conn.cursor()
182        cur2.execute("""
183            SELECT signature_ref, manuscript_shelfmark, hand_label,
184                   title, description, alchemical_element, alchemical_process,
185                   alchemical_framework, russell_page_ref
186            FROM folio_descriptions
187        """)
188        for fd in cur2.fetchall():
189            key = (fd[0], fd[1])
190            folio_descs[key] = {
191                'desc_title': fd[3], 'desc_text': fd[4],
192                'alch_element': fd[5], 'alch_process': fd[6],
193                'alch_framework': fd[7], 'russell_pages': fd[8],
194            }
195    except:
196        pass  # Table may not exist yet
197
198    entries = []
199    sigs = set()
200    for row in cur.fetchall():
201        sig = row[2]
202        ms = row[3]
203        entry = {
204            'ref_id': row[0], 'thesis_page': row[1],
205            'signature': sig, 'manuscript': ms,
206            'institution': row[4], 'city': row[5],
207            'context': (row[6] or '')[:600],
208            'marginal_text': row[7], 'chapter': row[8],
209            'image_file': row[9],
210            'image_path': row[10],
211            'folio_number': row[11], 'side': row[12],
212            'confidence': row[13] or 'PROVISIONAL',
213            'quire': row[14], 'leaf_in_quire': row[15],
214            'needs_review': bool(row[16]),
215            'hand_label': row[17], 'hand_attribution': row[18],
216            'is_alchemist': bool(row[19]) if row[19] is not None else False,
217        }
218        # Add folio description if available
219        fd = folio_descs.get((sig, ms)) or folio_descs.get((sig, None))
220        if fd:
221            entry['desc_title'] = fd['desc_title']
222            entry['desc_text'] = fd['desc_text']
223            entry['alch_element'] = fd['alch_element']
224
225        # Generate a brief description for every entry
226        parts = []
227        if entry.get('desc_title'):
228            parts.append(entry['desc_title'] + '.')
229        else:
230            # Build description from available data
231            ms_name = 'the BL copy' if ms == 'C.60.o.12' else 'the Siena copy'
232            if entry.get('hand_label') and entry.get('hand_attribution'):
233                hand_desc = f"Hand {entry['hand_label']}"
234                if entry['hand_attribution'] and entry['hand_attribution'] != 'Anonymous':
235                    hand_desc += f" ({entry['hand_attribution']})"
236                if entry.get('is_alchemist'):
237                    hand_desc += ', an alchemist'
238                parts.append(f"Annotated by {hand_desc} in {ms_name}.")
239            elif entry.get('hand_label'):
240                parts.append(f"Annotated by Hand {entry['hand_label']} in {ms_name}.")
241            else:
242                parts.append(f"A folio from {ms_name} referenced in Russell's thesis.")
243
244            if entry.get('marginal_text'):
245                mt = entry['marginal_text']
246                if len(mt) > 5:
247                    parts.append(f'Russell records the marginal note: "{mt[:80]}{"..." if len(mt) > 80 else ""}"')
248
249        entry['card_description'] = ' '.join(parts)
250        entries.append(entry)
251        sigs.add(sig)
252
253    data = {
254        'entries': entries,
255        'stats': {
256            'total_references': len(entries),
257            'unique_signatures': len(sigs),
258            'high_confidence_matches': sum(1 for e in entries if e['confidence'] == 'HIGH'),
259            'low_confidence_matches': sum(1 for e in entries if e['confidence'] == 'LOW'),
260            'needs_review': sum(1 for e in entries if e['needs_review']),
261        },
262        'provenance': {
263            'source': 'hp.db v2',
264            'note': 'BL C.60.o.12 matches are LOW confidence (1545 edition, unverified photo-folio mapping)',
265        },
266    }
267
268    out = SITE_DIR / 'data.json'
269    with open(out, 'w', encoding='utf-8') as f:
270        json.dump(data, f, indent=2, ensure_ascii=False)
271    print(f"  data.json: {len(entries)} entries")

278def build_scholars_pages(conn):
279    """Generate scholars.html and scholar/*.html from DB + summaries.json.
280
281    Uses scholar_overview, is_historical_subject, and scholar_works data
282    to build rich profiles per docs/SCHOLAR_SPEC.md.
283    """
284    # Load summaries for detailed content
285    summaries = []
286    if SUMMARIES_PATH.exists():
287        with open(SUMMARIES_PATH, encoding='utf-8') as f:
288            summaries = json.load(f)
289
290    # Index summaries by (author, title) for lookup
291    summary_lookup = {}
292    for s in summaries:
293        key = (s.get('author', ''), s.get('title', ''))
294        summary_lookup[key] = s
295
296    # Group summaries by author for backward compat
297    by_author_summaries = {}
298    for s in summaries:
299        author = s.get('author', 'Unknown')
300        by_author_summaries.setdefault(author, []).append(s)
301
302    cur = conn.cursor()
303
304    # Get all scholars with overview data
305    cur.execute("""
306        SELECT id, name, specialization, hp_focus, bio_notes,
307               scholar_overview, is_historical_subject,
308               needs_review, source_method, review_status
309        FROM scholars ORDER BY name
310    """)
311    all_scholars = cur.fetchall()
312
313    # Get scholar_works with bibliography data
314    cur.execute("""
315        SELECT sw.scholar_id, b.author, b.title, b.year, b.pub_type,
316               b.journal_or_publisher, sw.has_summary, sw.summary_source,
317               b.hp_relevance, b.in_collection
318        FROM scholar_works sw
319        JOIN bibliography b ON sw.bib_id = b.id
320        ORDER BY b.year DESC
321    """)
322    works_by_scholar = {}
323    for row in cur.fetchall():
324        works_by_scholar.setdefault(row[0], []).append(row)
325
326    scholar_cards = []
327    scholar_dir = SITE_DIR / 'scholar'
328    scholar_dir.mkdir(exist_ok=True)
329    pages_built = 0
330
331    for scholar in all_scholars:
332        (sid, name, specialization, hp_focus, bio_notes,
333         overview, is_hist, needs_rev, source_method, rev_status) = scholar
334
335        slug = slugify(name)
336        works = works_by_scholar.get(sid, [])
337        author_summaries = by_author_summaries.get(name, [])
338
339        # Collect topics from summaries
340        topics = set()
341        for p in author_summaries:
342            tc = p.get('topic_cluster', '')
343            if tc:
344                for t in tc.split(','):
345                    topics.add(t.strip())
346
347        badges = topic_badges_html(','.join(topics))
348        review_html = review_badge_html(needs_rev, source_method)
349
350        # Historical figure badge
351        hist_badge = ' <span class="review-status-badge review-badge-provisional">Historical Figure</span>' if is_hist else ''
352
353        # Overview preview for card (300 chars)
354        overview_text = overview or ''
355        overview_first_para = overview_text.split('\n\n')[0] if overview_text else ''
356        overview_preview = overview_first_para[:300]
357        if len(overview_first_para) > 300:
358            overview_preview = overview_preview.rsplit(' ', 1)[0] + '...'
359
360        # Work count
361        work_count = len(works) or len(author_summaries)
362        work_label = f"{work_count} work{'s' if work_count != 1 else ''}" if work_count else "No works catalogued"
363
364        # Card
365        overview_card_html = ''
366        if overview_preview:
367            overview_card_html = f"""
368                <div class="scholar-overview-preview">
369                    <p>{escape(overview_preview)}
370                    <a href="scholar/{slug}.html">Read more</a></p>
371                </div>"""
372
373        scholar_cards.append(f"""
374        <div class="scholar-card">
375            <h3><a href="scholar/{slug}.html">{escape(name)}</a>{hist_badge} {review_html}</h3>
376            <div class="scholar-meta">{work_label} {badges}</div>
377            {overview_card_html}
378        </div>""")
379
380        # === Detail page ===
381
382        # Full overview
383        overview_html = ''
384        if overview_text:
385            paras = overview_text.split('\n\n')
386            overview_html = ''.join(f'<p>{escape(p.strip())}</p>' for p in paras if p.strip())
387            overview_html = f'<div class="scholar-overview">{overview_html}</div>'
388
389        # Works in archive (with summaries)
390        archive_works_html = ''
391        works_with_summaries = [w for w in works if w[6]]  # has_summary = True
392        if not works_with_summaries and author_summaries:
393            # Fallback to summaries.json directly
394            for p in author_summaries:
395                tc = p.get('topic_cluster', '')
396                archive_works_html += f"""
397                <div class="paper-detail">
398                    <h4>{escape(p.get('title', ''))}</h4>
399                    <div class="paper-meta">{escape(p.get('journal', ''))} ({p.get('year', '?')})
400                        {topic_badges_html(tc)}</div>
401                    <div class="paper-summary-full"><p>{escape(p.get('summary', ''))}</p></div>
402                </div>"""
403        else:
404            for w in works_with_summaries:
405                _, bauthor, btitle, byear, bpubtype, bjournal, _, _, _, _ = w
406                # Find matching summary
407                summary_text = ''
408                for s in author_summaries:
409                    if s.get('title', '') == btitle:
410                        summary_text = s.get('summary', '')
411                        break
412                tc = ''
413                for s in author_summaries:
414                    if s.get('title', '') == btitle:
415                        tc = s.get('topic_cluster', '')
416                        break
417
418                archive_works_html += f"""
419                <div class="paper-detail">
420                    <h4>{escape(btitle or '')}</h4>
421                    <div class="paper-meta">{escape(bjournal or '')} ({byear or '?'}) [{bpubtype or ''}]
422                        {topic_badges_html(tc)}</div>
423                    <div class="paper-summary-full"><p>{escape(summary_text)}</p></div>
424                </div>"""
425
426        # Other known works (without summaries)
427        other_works_html = ''
428        works_without_summaries = [w for w in works if not w[6]]
429        for w in works_without_summaries:
430            _, bauthor, btitle, byear, bpubtype, bjournal, _, _, _, _ = w
431            other_works_html += f"""
432            <div class="paper-detail">
433                <h4>{escape(btitle or '')}</h4>
434                <div class="paper-meta">{escape(bjournal or '')} ({byear or '?'}) [{bpubtype or ''}]</div>
435                <p class="no-summary" style="color:var(--text-muted); font-style:italic; font-size:0.9rem">Summary not yet available.</p>
436            </div>"""
437
438        # Build sections
439        works_section = ''
440        if archive_works_html:
441            works_section += f'<h3>Works in Archive</h3>{archive_works_html}'
442        if other_works_html:
443            works_section += f'<h3>Other Known Works</h3>{other_works_html}'
444        if not works_section and not author_summaries:
445            works_section = '<p style="color:var(--text-muted)">No works catalogued yet.</p>'
446
447        # Provenance
448        provenance_html = f"""
449        <div class="provenance-section" style="margin-top:2rem; padding:1rem; background:var(--bg-card); border-radius:4px">
450            <h4 style="margin-top:0">Review Status / Provenance</h4>
451            <p>{review_status_badge(rev_status or 'DRAFT')} Source: {escape(source_method or 'LLM_ASSISTED')}</p>
452        </div>"""
453
454        detail_body = f"""
455        <div class="scholar-detail">
456            <p><a href="../scholars.html">&larr; All Scholars</a></p>
457            <h2>{escape(name)}{hist_badge} {review_html}</h2>
458            {overview_html}
459            {works_section}
460            {provenance_html}
461        </div>"""
462
463        detail_page = page_shell(name, detail_body, active_nav='scholars', depth=1)
464        (scholar_dir / f'{slug}.html').write_text(detail_page, encoding='utf-8')
465        pages_built += 1
466
467    # Build index page
468    # Count stats
469    modern_count = sum(1 for s in all_scholars if not s[6])
470    hist_count = sum(1 for s in all_scholars if s[6])
471
472    index_body = f"""
473        <div class="scholars-grid">
474            <div class="intro" style="margin-bottom:2rem">
475                <h2>Scholars of the <em>Hypnerotomachia</em></h2>
476                <p>The scholarship on the <em>Hypnerotomachia Poliphili</em> is wide-ranging
477                and interdisciplinary. It spans bibliography, book history, architecture,
478                garden studies, allegory, philology, reception history, emblematic reading,
479                and the history of interpretation. This section organizes that scholarly
480                landscape by person: {modern_count} modern scholars and {hist_count} historical
481                figures, with {len(summaries)} article and monograph summaries.</p>
482                <p>These pages are meant to make the field legible at a glance.
483                Where content is LLM-assisted or otherwise provisional, that status
484                remains visible.</p>
485            </div>
486            {''.join(scholar_cards)}
487        </div>"""
488
489    index_page = page_shell('Scholars', index_body, active_nav='scholars')
490    (SITE_DIR / 'scholars.html').write_text(index_page, encoding='utf-8')
491    print(f"  scholars.html + {pages_built} scholar pages")

1405def markdown_to_html(md_text):
1406    """Minimal markdown-to-HTML conversion for document display."""
1407    import re
1408    lines = md_text.split('\n')
1409    html_lines = []
1410    in_code = False
1411    in_list = False
1412    in_table = False
1413    table_header_done = False
1414
1415    for line in lines:
1416        # Code blocks
1417        if line.strip().startswith('```'):
1418            if in_code:
1419                html_lines.append('</code></pre>')
1420                in_code = False
1421            else:
1422                lang = line.strip()[3:]
1423                html_lines.append(f'<pre><code class="lang-{lang}">')
1424                in_code = True
1425            continue
1426        if in_code:
1427            html_lines.append(escape(line))
1428            continue
1429
1430        # Close list if needed
1431        if in_list and not line.strip().startswith(('-', '*', '1.', '2.', '3.', '4.', '5.', '6.', '7.', '8.', '9.')):
1432            html_lines.append('</ul>')
1433            in_list = False
1434
1435        # Tables
1436        if '|' in line and line.strip().startswith('|'):
1437            cells = [c.strip() for c in line.strip().split('|')[1:-1]]
1438            if all(set(c) <= set('- :') for c in cells):
1439                table_header_done = True
1440                continue
1441            if not in_table:
1442                html_lines.append('<table class="doc-table">')
1443                in_table = True
1444            tag = 'th' if not table_header_done else 'td'
1445            row = ''.join(f'<{tag}>{escape(c)}</{tag}>' for c in cells)
1446            html_lines.append(f'<tr>{row}</tr>')
1447            continue
1448        elif in_table:
1449            html_lines.append('</table>')
1450            in_table = False
1451            table_header_done = False
1452
1453        stripped = line.strip()
1454
1455        # Headings
1456        if stripped.startswith('### '):
1457            html_lines.append(f'<h4>{escape(stripped[4:])}</h4>')
1458        elif stripped.startswith('## '):
1459            html_lines.append(f'<h3>{escape(stripped[3:])}</h3>')
1460        elif stripped.startswith('# '):
1461            html_lines.append(f'<h2>{escape(stripped[2:])}</h2>')
1462        elif stripped.startswith('---'):
1463            html_lines.append('<hr>')
1464        elif stripped.startswith(('- ', '* ')):
1465            if not in_list:
1466                html_lines.append('<ul>')
1467                in_list = True
1468            html_lines.append(f'<li>{escape(stripped[2:])}</li>')
1469        elif stripped.startswith('>'):
1470            html_lines.append(f'<blockquote>{escape(stripped[1:].strip())}</blockquote>')
1471        elif stripped == '':
1472            html_lines.append('')
1473        else:
1474            # Apply inline formatting
1475            text = escape(stripped)
1476            # Bold
1477            text = re.sub(r'\*\*(.+?)\*\*', r'<strong>\1</strong>', text)
