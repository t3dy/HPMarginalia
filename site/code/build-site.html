31def slugify(text):
32    s = text.lower().strip()
33    s = re.sub(r'[^\w\s-]', '', s)
34    s = re.sub(r'[\s_]+', '-', s)
35    s = re.sub(r'-+', '-', s)
36    return s.strip('-')

155def export_data_json(conn):
156    """Export marginalia gallery data with confidence flags."""
157    cur = conn.cursor()
158    cur.execute("""
159        SELECT
160            a.id, a.thesis_page, a.signature_ref, m.shelfmark,
161            m.institution, m.city, dr.context_text, a.annotation_text,
162            a.thesis_chapter, i.filename, COALESCE(i.web_path, i.relative_path),
163            i.folio_number, i.side, mat.confidence,
164            sm.quire, sm.leaf_in_quire,
165            mat.needs_review as match_needs_review,
166            h.hand_label, h.attribution, h.is_alchemist,
167            a.annotation_type
168        FROM matches mat
169        JOIN annotations a ON mat.ref_id = a.id
170        JOIN images i ON mat.image_id = i.id
171        JOIN manuscripts m ON i.manuscript_id = m.id
172        LEFT JOIN dissertation_refs dr ON a.id = dr.id
173        LEFT JOIN signature_map sm ON LOWER(a.signature_ref) = LOWER(sm.signature)
174        LEFT JOIN annotator_hands h ON a.hand_id = h.id
175        WHERE i.page_type = 'PAGE'
176        GROUP BY a.signature_ref, i.filename
177        ORDER BY COALESCE(sm.folio_number, 999), a.thesis_page
178    """)
179
180    # Load folio descriptions for alchemist annotations
181    folio_descs = {}
182    try:
183        cur2 = conn.cursor()
184        cur2.execute("""
185            SELECT signature_ref, manuscript_shelfmark, hand_label,
186                   title, description, alchemical_element, alchemical_process,
187                   alchemical_framework, russell_page_ref
188            FROM folio_descriptions
189        """)
190        for fd in cur2.fetchall():
191            key = (fd[0], fd[1])
192            folio_descs[key] = {
193                'desc_title': fd[3], 'desc_text': fd[4],
194                'alch_element': fd[5], 'alch_process': fd[6],
195                'alch_framework': fd[7], 'russell_pages': fd[8],
196            }
197    except:
198        pass  # Table may not exist yet
199
200    entries = []
201    sigs = set()
202    for row in cur.fetchall():
203        sig = row[2]
204        ms = row[3]
205        entry = {
206            'ref_id': row[0], 'thesis_page': row[1],
207            'signature': sig, 'manuscript': ms,
208            'institution': row[4], 'city': row[5],
209            'context': (row[6] or '')[:600],
210            'marginal_text': row[7], 'chapter': row[8],
211            'image_file': row[9],
212            'image_path': row[10],
213            'folio_number': row[11], 'side': row[12],
214            'confidence': row[13] or 'PROVISIONAL',
215            'quire': row[14], 'leaf_in_quire': row[15],
216            'needs_review': bool(row[16]),
217            'hand_label': row[17], 'hand_attribution': row[18],
218            'is_alchemist': bool(row[19]) if row[19] is not None else False,
219            'annotation_type': row[20],
220        }
221        # Add folio description if available
222        fd = folio_descs.get((sig, ms)) or folio_descs.get((sig, None))
223        if fd:
224            entry['desc_title'] = fd['desc_title']
225            entry['desc_text'] = fd['desc_text']
226            entry['alch_element'] = fd['alch_element']
227
228        # Generate a brief description for every entry
229        parts = []
230        if entry.get('desc_title'):
231            parts.append(entry['desc_title'] + '.')
232        else:
233            # Build description from available data
234            ms_name = 'the BL copy' if ms == 'C.60.o.12' else 'the Siena copy'
235            if entry.get('hand_label') and entry.get('hand_attribution'):
236                hand_desc = f"Hand {entry['hand_label']}"
237                if entry['hand_attribution'] and entry['hand_attribution'] != 'Anonymous':
238                    hand_desc += f" ({entry['hand_attribution']})"
239                if entry.get('is_alchemist'):
240                    hand_desc += ', an alchemist'
241                parts.append(f"Annotated by {hand_desc} in {ms_name}.")
242            elif entry.get('hand_label'):
243                parts.append(f"Annotated by Hand {entry['hand_label']} in {ms_name}.")
244            else:
245                parts.append(f"A folio from {ms_name} referenced in Russell's thesis.")
246
247            if entry.get('marginal_text'):
248                mt = entry['marginal_text']
249                if len(mt) > 5:
250                    parts.append(f'Russell records the marginal note: "{mt[:80]}{"..." if len(mt) > 80 else ""}"')
251
252        entry['card_description'] = ' '.join(parts)
253        entries.append(entry)
254        sigs.add(sig)
255
256    data = {
257        'entries': entries,
258        'stats': {
259            'total_references': len(entries),
260            'unique_signatures': len(sigs),
261            'high_confidence_matches': sum(1 for e in entries if e['confidence'] == 'HIGH'),
262            'low_confidence_matches': sum(1 for e in entries if e['confidence'] == 'LOW'),
263            'needs_review': sum(1 for e in entries if e['needs_review']),
264        },
265        'provenance': {
266            'source': 'hp.db v2',
267            'note': 'BL C.60.o.12 matches are LOW confidence (1545 edition, unverified photo-folio mapping)',
268        },
269    }
270
271    out = SITE_DIR / 'data.json'
272    with open(out, 'w', encoding='utf-8') as f:
273        json.dump(data, f, indent=2, ensure_ascii=False)
274    print(f"  data.json: {len(entries)} entries")

281def build_scholars_pages(conn):
282    """Generate scholars.html and scholar/*.html from DB + summaries.json.
283
284    Uses scholar_overview, is_historical_subject, and scholar_works data
285    to build rich profiles per docs/SCHOLAR_SPEC.md.
286    """
287    # Load summaries for detailed content
288    summaries = []
289    if SUMMARIES_PATH.exists():
290        with open(SUMMARIES_PATH, encoding='utf-8') as f:
291            summaries = json.load(f)
292
293    # Index summaries by (author, title) for lookup
294    summary_lookup = {}
295    for s in summaries:
296        key = (s.get('author', ''), s.get('title', ''))
297        summary_lookup[key] = s
298
299    # Group summaries by author for backward compat
300    by_author_summaries = {}
301    for s in summaries:
302        author = s.get('author', 'Unknown')
303        by_author_summaries.setdefault(author, []).append(s)
304
305    cur = conn.cursor()
306
307    # Get all scholars with overview data
308    cur.execute("""
309        SELECT id, name, specialization, hp_focus, bio_notes,
310               scholar_overview, is_historical_subject,
311               needs_review, source_method, review_status
312        FROM scholars ORDER BY name
313    """)
314    all_scholars = cur.fetchall()
315
316    # Get scholar_works with bibliography data
317    cur.execute("""
318        SELECT sw.scholar_id, b.author, b.title, b.year, b.pub_type,
319               b.journal_or_publisher, sw.has_summary, sw.summary_source,
320               b.hp_relevance, b.in_collection
321        FROM scholar_works sw
322        JOIN bibliography b ON sw.bib_id = b.id
323        ORDER BY b.year DESC
324    """)
325    works_by_scholar = {}
326    for row in cur.fetchall():
327        works_by_scholar.setdefault(row[0], []).append(row)
328
329    scholar_cards = []
330    scholar_dir = SITE_DIR / 'scholar'
331    scholar_dir.mkdir(exist_ok=True)
332    pages_built = 0
333
334    for scholar in all_scholars:
335        (sid, name, specialization, hp_focus, bio_notes,
336         overview, is_hist, needs_rev, source_method, rev_status) = scholar
337
338        slug = slugify(name)
339        works = works_by_scholar.get(sid, [])
340        author_summaries = by_author_summaries.get(name, [])
341
342        # Collect topics from summaries
343        topics = set()
344        for p in author_summaries:
345            tc = p.get('topic_cluster', '')
346            if tc:
347                for t in tc.split(','):
348                    topics.add(t.strip())
349
350        badges = topic_badges_html(','.join(topics))
351        review_html = review_badge_html(needs_rev, source_method)
352
353        # Historical figure badge
354        hist_badge = ' <span class="review-status-badge review-badge-provisional">Historical Figure</span>' if is_hist else ''
355
356        # Overview preview for card (300 chars)
357        overview_text = overview or ''
358        overview_first_para = overview_text.split('\n\n')[0] if overview_text else ''
359        overview_preview = overview_first_para[:300]
360        if len(overview_first_para) > 300:
361            overview_preview = overview_preview.rsplit(' ', 1)[0] + '...'
362
363        # Work count
364        work_count = len(works) or len(author_summaries)
365        work_label = f"{work_count} work{'s' if work_count != 1 else ''}" if work_count else "No works catalogued"
366
367        # Card
368        overview_card_html = ''
369        if overview_preview:
370            overview_card_html = f"""
371                <div class="scholar-overview-preview">
372                    <p>{escape(overview_preview)}
373                    <a href="scholar/{slug}.html">Read more</a></p>
374                </div>"""
375
376        scholar_cards.append(f"""
377        <div class="scholar-card">
378            <h3><a href="scholar/{slug}.html">{escape(name)}</a>{hist_badge} {review_html}</h3>
379            <div class="scholar-meta">{work_label} {badges}</div>
380            {overview_card_html}
381        </div>""")
382
383        # === Detail page ===
384
385        # Full overview
386        overview_html = ''
387        if overview_text:
388            paras = overview_text.split('\n\n')
389            overview_html = ''.join(f'<p>{escape(p.strip())}</p>' for p in paras if p.strip())
390            overview_html = f'<div class="scholar-overview">{overview_html}</div>'
391
392        # Works in archive (with summaries)
393        archive_works_html = ''
394        works_with_summaries = [w for w in works if w[6]]  # has_summary = True
395        if not works_with_summaries and author_summaries:
396            # Fallback to summaries.json directly
397            for p in author_summaries:
398                tc = p.get('topic_cluster', '')
399                archive_works_html += f"""
400                <div class="paper-detail">
401                    <h4>{escape(p.get('title', ''))}</h4>
402                    <div class="paper-meta">{escape(p.get('journal', ''))} ({p.get('year', '?')})
403                        {topic_badges_html(tc)}</div>
404                    <div class="paper-summary-full"><p>{escape(p.get('summary', ''))}</p></div>
405                </div>"""
406        else:
407            for w in works_with_summaries:
408                _, bauthor, btitle, byear, bpubtype, bjournal, _, _, _, _ = w
409                # Find matching summary
410                summary_text = ''
411                for s in author_summaries:
412                    if s.get('title', '') == btitle:
413                        summary_text = s.get('summary', '')
414                        break
415                tc = ''
416                for s in author_summaries:
417                    if s.get('title', '') == btitle:
418                        tc = s.get('topic_cluster', '')
419                        break
420
421                archive_works_html += f"""
422                <div class="paper-detail">
423                    <h4>{escape(btitle or '')}</h4>
424                    <div class="paper-meta">{escape(bjournal or '')} ({byear or '?'}) [{bpubtype or ''}]
425                        {topic_badges_html(tc)}</div>
426                    <div class="paper-summary-full"><p>{escape(summary_text)}</p></div>
427                </div>"""
428
429        # Other known works (without summaries)
430        other_works_html = ''
431        works_without_summaries = [w for w in works if not w[6]]
432        for w in works_without_summaries:
433            _, bauthor, btitle, byear, bpubtype, bjournal, _, _, _, _ = w
434            other_works_html += f"""
435            <div class="paper-detail">
436                <h4>{escape(btitle or '')}</h4>
437                <div class="paper-meta">{escape(bjournal or '')} ({byear or '?'}) [{bpubtype or ''}]</div>
438                <p class="no-summary" style="color:var(--text-muted); font-style:italic; font-size:0.9rem">Summary not yet available.</p>
439            </div>"""
440
441        # Build sections
442        works_section = ''
443        if archive_works_html:
444            works_section += f'<h3>Works in Archive</h3>{archive_works_html}'
445        if other_works_html:
446            works_section += f'<h3>Other Known Works</h3>{other_works_html}'
447        if not works_section and not author_summaries:
448            works_section = '<p style="color:var(--text-muted)">No works catalogued yet.</p>'
449
450        # Provenance
451        provenance_html = f"""
452        <div class="provenance-section" style="margin-top:2rem; padding:1rem; background:var(--bg-card); border-radius:4px">
453            <h4 style="margin-top:0">Review Status / Provenance</h4>
454            <p>{review_status_badge(rev_status or 'DRAFT')} Source: {escape(source_method or 'LLM_ASSISTED')}</p>
455        </div>"""
456
457        detail_body = f"""
458        <div class="scholar-detail">
459            <p><a href="../scholars.html">&larr; All Scholars</a></p>
460            <h2>{escape(name)}{hist_badge} {review_html}</h2>
461            {overview_html}
462            {works_section}
463            {provenance_html}
464        </div>"""
465
466        detail_page = page_shell(name, detail_body, active_nav='scholars', depth=1)
467        (scholar_dir / f'{slug}.html').write_text(detail_page, encoding='utf-8')
468        pages_built += 1
469
470    # Build index page
471    # Count stats
472    modern_count = sum(1 for s in all_scholars if not s[6])
473    hist_count = sum(1 for s in all_scholars if s[6])
474
475    index_body = f"""
476        <div class="scholars-grid">
477            <div class="intro" style="margin-bottom:2rem">
478                <h2>Scholars of the <em>Hypnerotomachia</em></h2>
479                <p>The scholarship on the <em>Hypnerotomachia Poliphili</em> is wide-ranging
480                and interdisciplinary. It spans bibliography, book history, architecture,
481                garden studies, allegory, philology, reception history, emblematic reading,
482                and the history of interpretation. This section organizes that scholarly
483                landscape by person: {modern_count} modern scholars and {hist_count} historical
484                figures, with {len(summaries)} article and monograph summaries.</p>
485                <p>These pages are meant to make the field legible at a glance.
486                Where content is LLM-assisted or otherwise provisional, that status
487                remains visible.</p>
488            </div>
489            {''.join(scholar_cards)}
490        </div>"""
491
492    index_page = page_shell('Scholars', index_body, active_nav='scholars')
493    (SITE_DIR / 'scholars.html').write_text(index_page, encoding='utf-8')
494    print(f"  scholars.html + {pages_built} scholar pages")

1637def markdown_to_html(md_text):
1638    """Minimal markdown-to-HTML conversion for document display."""
1639    import re
1640    lines = md_text.split('\n')
1641    html_lines = []
1642    in_code = False
1643    in_list = False
1644    in_table = False
1645    table_header_done = False
1646
1647    for line in lines:
1648        # Code blocks
1649        if line.strip().startswith('```'):
1650            if in_code:
1651                html_lines.append('</code></pre>')
1652                in_code = False
1653            else:
1654                lang = line.strip()[3:]
1655                html_lines.append(f'<pre><code class="lang-{lang}">')
1656                in_code = True
1657            continue
1658        if in_code:
1659            html_lines.append(escape(line))
1660            continue
1661
1662        # Close list if needed
1663        if in_list and not line.strip().startswith(('-', '*', '1.', '2.', '3.', '4.', '5.', '6.', '7.', '8.', '9.')):
1664            html_lines.append('</ul>')
1665            in_list = False
1666
1667        # Tables
1668        if '|' in line and line.strip().startswith('|'):
1669            cells = [c.strip() for c in line.strip().split('|')[1:-1]]
1670            if all(set(c) <= set('- :') for c in cells):
1671                table_header_done = True
1672                continue
1673            if not in_table:
1674                html_lines.append('<table class="doc-table">')
1675                in_table = True
1676            tag = 'th' if not table_header_done else 'td'
1677            row = ''.join(f'<{tag}>{escape(c)}</{tag}>' for c in cells)
1678            html_lines.append(f'<tr>{row}</tr>')
1679            continue
1680        elif in_table:
1681            html_lines.append('</table>')
1682            in_table = False
1683            table_header_done = False
1684
1685        stripped = line.strip()
1686
1687        # Headings
1688        if stripped.startswith('### '):
1689            html_lines.append(f'<h4>{escape(stripped[4:])}</h4>')
1690        elif stripped.startswith('## '):
1691            html_lines.append(f'<h3>{escape(stripped[3:])}</h3>')
1692        elif stripped.startswith('# '):
1693            html_lines.append(f'<h2>{escape(stripped[2:])}</h2>')
1694        elif stripped.startswith('---'):
1695            html_lines.append('<hr>')
1696        elif stripped.startswith(('- ', '* ')):
1697            if not in_list:
1698                html_lines.append('<ul>')
1699                in_list = True
1700            html_lines.append(f'<li>{escape(stripped[2:])}</li>')
1701        elif stripped.startswith('>'):
1702            html_lines.append(f'<blockquote>{escape(stripped[1:].strip())}</blockquote>')
1703        elif stripped == '':
1704            html_lines.append('')
1705        else:
1706            # Apply inline formatting
1707            text = escape(stripped)
1708            # Bold
1709            text = re.sub(r'\*\*(.+?)\*\*', r'<strong>\1</strong>', text)
1710            # Italic
1711            text = re.sub(r'\*(.+?)\*', r'<em>\1</em>', text)
1712            # Inline code
1713            text = re.sub(r'`(.+?)`', r'<code>\1</code>', text)
1714            html_lines.append(f'<p>{text}</p>')
1715
1716    if in_list:
1717        html_lines.append('</ul>')
1718    if in_table:
1719        html_lines.append('</table>')
1720    if in_code:
1721        html_lines.append('</code></pre>')
1722
1723    return '\n'.join(html_lines)
