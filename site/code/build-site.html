31def slugify(text):
32    s = text.lower().strip()
33    s = re.sub(r'[^\w\s-]', '', s)
34    s = re.sub(r'[\s_]+', '-', s)
35    s = re.sub(r'-+', '-', s)
36    return s.strip('-')

153def export_data_json(conn):
154    """Export marginalia gallery data with confidence flags."""
155    cur = conn.cursor()
156    cur.execute("""
157        SELECT
158            r.id, r.thesis_page, r.signature_ref, m.shelfmark,
159            m.institution, m.city, r.context_text, r.marginal_text,
160            r.chapter_num, i.filename, i.relative_path,
161            i.folio_number, i.side, mat.confidence,
162            sm.quire, sm.leaf_in_quire,
163            mat.needs_review as match_needs_review,
164            h.hand_label, h.attribution, h.is_alchemist
165        FROM matches mat
166        JOIN dissertation_refs r ON mat.ref_id = r.id
167        JOIN images i ON mat.image_id = i.id
168        JOIN manuscripts m ON i.manuscript_id = m.id
169        LEFT JOIN signature_map sm ON LOWER(r.signature_ref) = LOWER(sm.signature)
170        LEFT JOIN annotator_hands h ON r.hand_id = h.id
171        WHERE i.page_type = 'PAGE'
172        GROUP BY r.signature_ref, i.filename
173        ORDER BY COALESCE(sm.folio_number, 999), r.thesis_page
174    """)
175
176    # Load folio descriptions for alchemist annotations
177    folio_descs = {}
178    try:
179        cur2 = conn.cursor()
180        cur2.execute("""
181            SELECT signature_ref, manuscript_shelfmark, hand_label,
182                   title, description, alchemical_element, alchemical_process,
183                   alchemical_framework, russell_page_ref
184            FROM folio_descriptions
185        """)
186        for fd in cur2.fetchall():
187            key = (fd[0], fd[1])
188            folio_descs[key] = {
189                'desc_title': fd[3], 'desc_text': fd[4],
190                'alch_element': fd[5], 'alch_process': fd[6],
191                'alch_framework': fd[7], 'russell_pages': fd[8],
192            }
193    except:
194        pass  # Table may not exist yet
195
196    entries = []
197    sigs = set()
198    for row in cur.fetchall():
199        sig = row[2]
200        ms = row[3]
201        entry = {
202            'ref_id': row[0], 'thesis_page': row[1],
203            'signature': sig, 'manuscript': ms,
204            'institution': row[4], 'city': row[5],
205            'context': (row[6] or '')[:600],
206            'marginal_text': row[7], 'chapter': row[8],
207            'image_file': row[9],
208            'image_path': row[10],
209            'folio_number': row[11], 'side': row[12],
210            'confidence': row[13] or 'PROVISIONAL',
211            'quire': row[14], 'leaf_in_quire': row[15],
212            'needs_review': bool(row[16]),
213            'hand_label': row[17], 'hand_attribution': row[18],
214            'is_alchemist': bool(row[19]) if row[19] is not None else False,
215        }
216        # Add folio description if available
217        fd = folio_descs.get((sig, ms)) or folio_descs.get((sig, None))
218        if fd:
219            entry['desc_title'] = fd['desc_title']
220            entry['desc_text'] = fd['desc_text']
221            entry['alch_element'] = fd['alch_element']
222        entries.append(entry)
223        sigs.add(sig)
224
225    data = {
226        'entries': entries,
227        'stats': {
228            'total_references': len(entries),
229            'unique_signatures': len(sigs),
230            'high_confidence_matches': sum(1 for e in entries if e['confidence'] == 'HIGH'),
231            'low_confidence_matches': sum(1 for e in entries if e['confidence'] == 'LOW'),
232            'needs_review': sum(1 for e in entries if e['needs_review']),
233        },
234        'provenance': {
235            'source': 'hp.db v2',
236            'note': 'BL C.60.o.12 matches are LOW confidence (1545 edition, unverified photo-folio mapping)',
237        },
238    }
239
240    out = SITE_DIR / 'data.json'
241    with open(out, 'w', encoding='utf-8') as f:
242        json.dump(data, f, indent=2, ensure_ascii=False)
243    print(f"  data.json: {len(entries)} entries")

250def build_scholars_pages(conn):
251    """Generate scholars.html and scholar/*.html from DB + summaries.json.
252
253    Uses scholar_overview, is_historical_subject, and scholar_works data
254    to build rich profiles per docs/SCHOLAR_SPEC.md.
255    """
256    # Load summaries for detailed content
257    summaries = []
258    if SUMMARIES_PATH.exists():
259        with open(SUMMARIES_PATH, encoding='utf-8') as f:
260            summaries = json.load(f)
261
262    # Index summaries by (author, title) for lookup
263    summary_lookup = {}
264    for s in summaries:
265        key = (s.get('author', ''), s.get('title', ''))
266        summary_lookup[key] = s
267
268    # Group summaries by author for backward compat
269    by_author_summaries = {}
270    for s in summaries:
271        author = s.get('author', 'Unknown')
272        by_author_summaries.setdefault(author, []).append(s)
273
274    cur = conn.cursor()
275
276    # Get all scholars with overview data
277    cur.execute("""
278        SELECT id, name, specialization, hp_focus, bio_notes,
279               scholar_overview, is_historical_subject,
280               needs_review, source_method, review_status
281        FROM scholars ORDER BY name
282    """)
283    all_scholars = cur.fetchall()
284
285    # Get scholar_works with bibliography data
286    cur.execute("""
287        SELECT sw.scholar_id, b.author, b.title, b.year, b.pub_type,
288               b.journal_or_publisher, sw.has_summary, sw.summary_source,
289               b.hp_relevance, b.in_collection
290        FROM scholar_works sw
291        JOIN bibliography b ON sw.bib_id = b.id
292        ORDER BY b.year DESC
293    """)
294    works_by_scholar = {}
295    for row in cur.fetchall():
296        works_by_scholar.setdefault(row[0], []).append(row)
297
298    scholar_cards = []
299    scholar_dir = SITE_DIR / 'scholar'
300    scholar_dir.mkdir(exist_ok=True)
301    pages_built = 0
302
303    for scholar in all_scholars:
304        (sid, name, specialization, hp_focus, bio_notes,
305         overview, is_hist, needs_rev, source_method, rev_status) = scholar
306
307        slug = slugify(name)
308        works = works_by_scholar.get(sid, [])
309        author_summaries = by_author_summaries.get(name, [])
310
311        # Collect topics from summaries
312        topics = set()
313        for p in author_summaries:
314            tc = p.get('topic_cluster', '')
315            if tc:
316                for t in tc.split(','):
317                    topics.add(t.strip())
318
319        badges = topic_badges_html(','.join(topics))
320        review_html = review_badge_html(needs_rev, source_method)
321
322        # Historical figure badge
323        hist_badge = ' <span class="review-status-badge review-badge-provisional">Historical Figure</span>' if is_hist else ''
324
325        # Overview preview for card (300 chars)
326        overview_text = overview or ''
327        overview_first_para = overview_text.split('\n\n')[0] if overview_text else ''
328        overview_preview = overview_first_para[:300]
329        if len(overview_first_para) > 300:
330            overview_preview = overview_preview.rsplit(' ', 1)[0] + '...'
331
332        # Work count
333        work_count = len(works) or len(author_summaries)
334        work_label = f"{work_count} work{'s' if work_count != 1 else ''}" if work_count else "No works catalogued"
335
336        # Card
337        overview_card_html = ''
338        if overview_preview:
339            overview_card_html = f"""
340                <div class="scholar-overview-preview">
341                    <p>{escape(overview_preview)}
342                    <a href="scholar/{slug}.html">Read more</a></p>
343                </div>"""
344
345        scholar_cards.append(f"""
346        <div class="scholar-card">
347            <h3><a href="scholar/{slug}.html">{escape(name)}</a>{hist_badge} {review_html}</h3>
348            <div class="scholar-meta">{work_label} {badges}</div>
349            {overview_card_html}
350        </div>""")
351
352        # === Detail page ===
353
354        # Full overview
355        overview_html = ''
356        if overview_text:
357            paras = overview_text.split('\n\n')
358            overview_html = ''.join(f'<p>{escape(p.strip())}</p>' for p in paras if p.strip())
359            overview_html = f'<div class="scholar-overview">{overview_html}</div>'
360
361        # Works in archive (with summaries)
362        archive_works_html = ''
363        works_with_summaries = [w for w in works if w[6]]  # has_summary = True
364        if not works_with_summaries and author_summaries:
365            # Fallback to summaries.json directly
366            for p in author_summaries:
367                tc = p.get('topic_cluster', '')
368                archive_works_html += f"""
369                <div class="paper-detail">
370                    <h4>{escape(p.get('title', ''))}</h4>
371                    <div class="paper-meta">{escape(p.get('journal', ''))} ({p.get('year', '?')})
372                        {topic_badges_html(tc)}</div>
373                    <div class="paper-summary-full"><p>{escape(p.get('summary', ''))}</p></div>
374                </div>"""
375        else:
376            for w in works_with_summaries:
377                _, bauthor, btitle, byear, bpubtype, bjournal, _, _, _, _ = w
378                # Find matching summary
379                summary_text = ''
380                for s in author_summaries:
381                    if s.get('title', '') == btitle:
382                        summary_text = s.get('summary', '')
383                        break
384                tc = ''
385                for s in author_summaries:
386                    if s.get('title', '') == btitle:
387                        tc = s.get('topic_cluster', '')
388                        break
389
390                archive_works_html += f"""
391                <div class="paper-detail">
392                    <h4>{escape(btitle or '')}</h4>
393                    <div class="paper-meta">{escape(bjournal or '')} ({byear or '?'}) [{bpubtype or ''}]
394                        {topic_badges_html(tc)}</div>
395                    <div class="paper-summary-full"><p>{escape(summary_text)}</p></div>
396                </div>"""
397
398        # Other known works (without summaries)
399        other_works_html = ''
400        works_without_summaries = [w for w in works if not w[6]]
401        for w in works_without_summaries:
402            _, bauthor, btitle, byear, bpubtype, bjournal, _, _, _, _ = w
403            other_works_html += f"""
404            <div class="paper-detail">
405                <h4>{escape(btitle or '')}</h4>
406                <div class="paper-meta">{escape(bjournal or '')} ({byear or '?'}) [{bpubtype or ''}]</div>
407                <p class="no-summary" style="color:var(--text-muted); font-style:italic; font-size:0.9rem">Summary not yet available.</p>
408            </div>"""
409
410        # Build sections
411        works_section = ''
412        if archive_works_html:
413            works_section += f'<h3>Works in Archive</h3>{archive_works_html}'
414        if other_works_html:
415            works_section += f'<h3>Other Known Works</h3>{other_works_html}'
416        if not works_section and not author_summaries:
417            works_section = '<p style="color:var(--text-muted)">No works catalogued yet.</p>'
418
419        # Provenance
420        provenance_html = f"""
421        <div class="provenance-section" style="margin-top:2rem; padding:1rem; background:var(--bg-card); border-radius:4px">
422            <h4 style="margin-top:0">Review Status / Provenance</h4>
423            <p>{review_status_badge(rev_status or 'DRAFT')} Source: {escape(source_method or 'LLM_ASSISTED')}</p>
424        </div>"""
425
426        detail_body = f"""
427        <div class="scholar-detail">
428            <p><a href="../scholars.html">&larr; All Scholars</a></p>
429            <h2>{escape(name)}{hist_badge} {review_html}</h2>
430            {overview_html}
431            {works_section}
432            {provenance_html}
433        </div>"""
434
435        detail_page = page_shell(name, detail_body, active_nav='scholars', depth=1)
436        (scholar_dir / f'{slug}.html').write_text(detail_page, encoding='utf-8')
437        pages_built += 1
438
439    # Build index page
440    # Count stats
441    modern_count = sum(1 for s in all_scholars if not s[6])
442    hist_count = sum(1 for s in all_scholars if s[6])
443
444    index_body = f"""
445        <div class="scholars-grid">
446            <div class="intro" style="margin-bottom:2rem">
447                <h2>Scholars of the <em>Hypnerotomachia</em></h2>
448                <p>The scholarship on the <em>Hypnerotomachia Poliphili</em> is wide-ranging
449                and interdisciplinary. It spans bibliography, book history, architecture,
450                garden studies, allegory, philology, reception history, emblematic reading,
451                and the history of interpretation. This section organizes that scholarly
452                landscape by person: {modern_count} modern scholars and {hist_count} historical
453                figures, with {len(summaries)} article and monograph summaries.</p>
454                <p>These pages are meant to make the field legible at a glance.
455                Where content is LLM-assisted or otherwise provisional, that status
456                remains visible.</p>
457            </div>
458            {''.join(scholar_cards)}
459        </div>"""
460
461    index_page = page_shell('Scholars', index_body, active_nav='scholars')
462    (SITE_DIR / 'scholars.html').write_text(index_page, encoding='utf-8')
463    print(f"  scholars.html + {pages_built} scholar pages")

1273def markdown_to_html(md_text):
1274    """Minimal markdown-to-HTML conversion for document display."""
1275    import re
1276    lines = md_text.split('\n')
1277    html_lines = []
1278    in_code = False
1279    in_list = False
1280    in_table = False
1281    table_header_done = False
1282
1283    for line in lines:
1284        # Code blocks
1285        if line.strip().startswith('```'):
1286            if in_code:
1287                html_lines.append('</code></pre>')
1288                in_code = False
1289            else:
1290                lang = line.strip()[3:]
1291                html_lines.append(f'<pre><code class="lang-{lang}">')
1292                in_code = True
1293            continue
1294        if in_code:
1295            html_lines.append(escape(line))
1296            continue
1297
1298        # Close list if needed
1299        if in_list and not line.strip().startswith(('-', '*', '1.', '2.', '3.', '4.', '5.', '6.', '7.', '8.', '9.')):
1300            html_lines.append('</ul>')
1301            in_list = False
1302
1303        # Tables
1304        if '|' in line and line.strip().startswith('|'):
1305            cells = [c.strip() for c in line.strip().split('|')[1:-1]]
1306            if all(set(c) <= set('- :') for c in cells):
1307                table_header_done = True
1308                continue
1309            if not in_table:
1310                html_lines.append('<table class="doc-table">')
1311                in_table = True
1312            tag = 'th' if not table_header_done else 'td'
1313            row = ''.join(f'<{tag}>{escape(c)}</{tag}>' for c in cells)
1314            html_lines.append(f'<tr>{row}</tr>')
1315            continue
1316        elif in_table:
1317            html_lines.append('</table>')
1318            in_table = False
1319            table_header_done = False
1320
1321        stripped = line.strip()
1322
1323        # Headings
1324        if stripped.startswith('### '):
1325            html_lines.append(f'<h4>{escape(stripped[4:])}</h4>')
1326        elif stripped.startswith('## '):
1327            html_lines.append(f'<h3>{escape(stripped[3:])}</h3>')
1328        elif stripped.startswith('# '):
1329            html_lines.append(f'<h2>{escape(stripped[2:])}</h2>')
1330        elif stripped.startswith('---'):
1331            html_lines.append('<hr>')
1332        elif stripped.startswith(('- ', '* ')):
1333            if not in_list:
1334                html_lines.append('<ul>')
1335                in_list = True
1336            html_lines.append(f'<li>{escape(stripped[2:])}</li>')
1337        elif stripped.startswith('>'):
1338            html_lines.append(f'<blockquote>{escape(stripped[1:].strip())}</blockquote>')
1339        elif stripped == '':
1340            html_lines.append('')
1341        else:
1342            # Apply inline formatting
1343            text = escape(stripped)
1344            # Bold
1345            text = re.sub(r'\*\*(.+?)\*\*', r'<strong>\1</strong>', text)
1346            # Italic
1347            text = re.sub(r'\*(.+?)\*', r'<em>\1</em>', text)
1348            # Inline code
1349            text = re.sub(r'`(.+?)`', r'<code>\1</code>', text)
1350            html_lines.append(f'<p>{text}</p>')
1351
1352    if in_list:
1353        html_lines.append('</ul>')
1354    if in_table:
1355        html_lines.append('</table>')
1356    if in_code:
1357        html_lines.append('</code></pre>')
1358
1359    return '\n'.join(html_lines)
