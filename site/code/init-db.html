3import sqlite3
4import os
5import re
6from pathlib import Path

8BASE_DIR = Path(__file__).resolve().parent.parent
9DB_PATH = BASE_DIR / "db" / "hp.db"

11SCHEMA = """
12CREATE TABLE IF NOT EXISTS documents (
13    id INTEGER PRIMARY KEY,
14    filename TEXT NOT NULL,
15    title TEXT,
16    author TEXT,
17    year INTEGER,
18    doc_type TEXT CHECK(doc_type IN ('PRIMARY_TEXT','DISSERTATION','SCHOLARSHIP','PRESENTATION')),
19    page_count INTEGER,
20    file_size_bytes INTEGER,
21    has_selectable_text BOOLEAN
22);
23
24CREATE TABLE IF NOT EXISTS manuscripts (
25    id INTEGER PRIMARY KEY,
26    shelfmark TEXT NOT NULL UNIQUE,
27    institution TEXT,
28    city TEXT,
29    description TEXT,
30    image_count INTEGER,
31    image_dir TEXT
32);
33
34CREATE TABLE IF NOT EXISTS images (
35    id INTEGER PRIMARY KEY,
36    manuscript_id INTEGER REFERENCES manuscripts(id),
37    filename TEXT NOT NULL,
38    folio_number TEXT,
39    side TEXT CHECK(side IN ('r','v',NULL)),
40    page_type TEXT CHECK(page_type IN ('PAGE','MARGINALIA_DETAIL','COVER','GUARD','OTHER')),
41    sort_order INTEGER,
42    relative_path TEXT NOT NULL
43);
44
45CREATE TABLE IF NOT EXISTS signature_map (
46    id INTEGER PRIMARY KEY,
47    signature TEXT NOT NULL UNIQUE,
48    folio_number INTEGER,
49    side TEXT CHECK(side IN ('r','v')),
50    quire TEXT,

52);
53
54CREATE TABLE IF NOT EXISTS dissertation_refs (
55    id INTEGER PRIMARY KEY,

57    signature_ref TEXT,
58    manuscript_shelfmark TEXT,
59    context_text TEXT,
60    marginal_text TEXT,

65
66CREATE TABLE IF NOT EXISTS matches (
67    id INTEGER PRIMARY KEY,

74
75CREATE TABLE IF NOT EXISTS schema_version (
76    version INTEGER PRIMARY KEY,
77    created_at TEXT DEFAULT (datetime('now'))

81# Known document metadata extracted from filenames
82DOC_CLASSIFICATIONS = {
83    'PRIMARY_TEXT': [

96def classify_doc(filename):
97    for doc_type, patterns in DOC_CLASSIFICATIONS.items():
98        for pat in patterns:
99            if pat in filename:
100                return doc_type
101    return 'SCHOLARSHIP'

104def extract_author_title(filename):
105    """Best-effort extraction of author and title from filename."""
106    name = Path(filename).stem
107    # Remove common suffixes
108    name = re.sub(r'\s*\(\d+\)\s*$', '', name)
109    # Try splitting on known patterns
110    parts = name.split(' - ')
111    if len(parts) == 2:
112        return parts[0].strip(), parts[1].strip()
113    # For space-separated names, take first few words as author guess
114    return None, name

117def populate_documents(conn):
118    """Scan root dir for PDFs and PPTX files."""
119    cur = conn.cursor()
120    extensions = {'.pdf', '.pptx'}
121    count = 0
122    for f in sorted(BASE_DIR.iterdir()):
123        if f.suffix.lower() in extensions and f.is_file():
124            author, title = extract_author_title(f.name)
125            doc_type = classify_doc(f.name)
126            size = f.stat().st_size
127            cur.execute(
128                """INSERT OR IGNORE INTO documents
129                   (filename, title, author, doc_type, file_size_bytes)
130                   VALUES (?, ?, ?, ?, ?)""",
131                (f.name, title, author, doc_type, size)
132            )
133            count += 1
134    conn.commit()
135    print(f"  Cataloged {count} documents")

161        cur.execute(

171def main():
172    print(f"Initializing database at {DB_PATH}")
173    conn = sqlite3.connect(DB_PATH)
174
175    print("Creating schema...")
176    conn.executescript(SCHEMA)
177    conn.execute("INSERT OR IGNORE INTO schema_version (version) VALUES (1)")
178    conn.commit()
179
180    print("Populating documents...")
181    populate_documents(conn)
182
183    print("Populating manuscripts...")
184    populate_manuscripts(conn)
185
186    conn.close()
187    print("Done.")

About This Project
