212def main():
213    conn = sqlite3.connect(DB_PATH)
214    cur = conn.cursor()
215
216    print("Seeding timeline events (v2)...")
217    inserted = 0
218    for e in EVENTS:
219        # Check for existing event with same year+title
220        cur.execute("SELECT id FROM timeline_events WHERE year = ? AND title = ?",
221                    (e["year"], e["title"]))
222        if cur.fetchone():
223            continue
224
225        cur.execute("""
226            INSERT INTO timeline_events
227                (year, event_type, title, description, category, medium,
228                 location, confidence, needs_review, source_method)
229            VALUES (?, ?, ?, ?, ?, ?, ?, ?, 1, 'LLM_ASSISTED')
230        """, (e["year"], e["event_type"], e["title"], e.get("description", ""),
231              e.get("category", ""), e.get("medium", ""),
232              e.get("location", ""), e.get("confidence", "MEDIUM")))
233        inserted += cur.rowcount
234
235    conn.commit()
236
237    cur.execute("SELECT COUNT(*) FROM timeline_events")
238    total = cur.fetchone()[0]
239    cur.execute("SELECT event_type, COUNT(*) FROM timeline_events GROUP BY event_type ORDER BY COUNT(*) DESC")
240    print(f"\nTotal events: {total}")
241    for row in cur.fetchall():
242        print(f"  {row[0]}: {row[1]}")
243
244    conn.close()
245    print(f"\nInserted {inserted} new events. Done.")
246
247
248if __name__ == "__main__":
249    main()

About This Project

A digital humanities project documenting the readership and marginalia of the 1499 Aldine Hypnerotomachia Poliphili, based on James Russell's PhD thesis (Durham, 2014).
