206def main():
207    conn = sqlite3.connect(DB_PATH)
208    cur = conn.cursor()
209
210    print("Seeding HP entity dictionary terms (v3)...")
211    inserted = 0
212    for t in TERMS:
213        cur.execute("""
214            INSERT OR IGNORE INTO dictionary_terms
215                (slug, label, category, definition_short, definition_long,
216                 source_basis, review_status, needs_review, source_method)
217            VALUES (?, ?, ?, ?, ?, ?, 'DRAFT', 1, 'LLM_ASSISTED')
218        """, (t["slug"], t["label"], t["category"],
219              t["definition_short"], t["definition_long"],
220              t.get("source_basis", "")))
221        inserted += cur.rowcount
222    conn.commit()
223    print(f"  Inserted {inserted} new terms ({len(TERMS)} defined)")
224
225    print("Creating term links...")
226    linked = 0
227    for slug_a, slug_b, link_type in LINKS:
228        cur.execute("SELECT id FROM dictionary_terms WHERE slug=?", (slug_a,))
229        a = cur.fetchone()
230        cur.execute("SELECT id FROM dictionary_terms WHERE slug=?", (slug_b,))
231        b = cur.fetchone()
232        if a and b:
233            cur.execute("""
234                INSERT OR IGNORE INTO dictionary_term_links
235                    (term_id, linked_term_id, link_type)
236                VALUES (?, ?, ?)
237            """, (a[0], b[0], link_type))
238            linked += cur.rowcount
239            if link_type == "RELATED":
240                cur.execute("""
241                    INSERT OR IGNORE INTO dictionary_term_links
242                        (term_id, linked_term_id, link_type)
243                    VALUES (?, ?, ?)
244                """, (b[0], a[0], link_type))
245                linked += cur.rowcount
246    conn.commit()
247    print(f"  Created {linked} links")
248
249    cur.execute("SELECT COUNT(*) FROM dictionary_terms")
250    print(f"\nTotal terms: {cur.fetchone()[0]}")
251    cur.execute("SELECT category, COUNT(*) FROM dictionary_terms GROUP BY category ORDER BY COUNT(*) DESC")
252    for row in cur.fetchall():
253        print(f"  {row[0]}: {row[1]}")
254
255    conn.close()
256    print("\nDone.")

About This Project

A digital scholarship platform exploring the alchemical annotations and wider marginalia of the 1499 Hypnerotomachia Poliphili, based on James Russell's PhD thesis (Durham, 2014).
