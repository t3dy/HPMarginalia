11def parse_bl_filename(filename):
12    """Parse BL C.60.o.12 image filenames.
13
14    Patterns:
15      C_60_o_12-NNN.jpg  -> sequential page scan
16      BL HP NN.jpg        -> marginalia detail photo
17      BL 2.1.jpg, BL2.jpg -> supplementary photos
18    """
19    # Sequential page scans
20    m = re.match(r'C_60_o_12-(\d{3})\.jpg', filename)
21    if m:
22        seq = int(m.group(1))
23        return {
24            'folio_number': str(seq),
25            'side': None,  # not encoded in filename
26            'page_type': 'PAGE',
27            'sort_order': seq,
28        }
29
30    # Marginalia detail photos
31    m = re.match(r'BL\s*HP\s*(\d+)\.jpg', filename)
32    if m:
33        num = int(m.group(1))
34        return {
35            'folio_number': None,
36            'side': None,
37            'page_type': 'MARGINALIA_DETAIL',
38            'sort_order': 1000 + num,
39        }
40
41    # Other BL supplementary
42    m = re.match(r'BL\s*[\d.]+\.jpg', filename)
43    if m:
44        return {
45            'folio_number': None,
46            'side': None,
47            'page_type': 'OTHER',
48            'sort_order': 2000,
49        }
50
51    return {
52        'folio_number': None,
53        'side': None,
54        'page_type': 'OTHER',
55        'sort_order': 9999,
56    }

59def parse_siena_filename(filename):
60    """Parse Siena O.III.38 image filenames.
61
62    Patterns:
63      O.III.38_NNNNr.jpg / O.III.38_NNNNv.jpg  -> folio recto/verso
64      O.III.38_000antcop.jpg                     -> front cover
65      O.III.38_000fdg1r.jpg                      -> guard leaf
66      O.III.38_postcop.jpg                       -> back cover
67      Siena HP N.jpg / Siena Hp N.jpg            -> marginalia detail
68    """
69    # Standard folio pages
70    m = re.match(r'O\.III\.38_(\d{4})([rv])\.jpg', filename)
71    if m:
72        folio = int(m.group(1))
73        side = m.group(2)
74        sort_order = folio * 2 + (0 if side == 'r' else 1)
75        return {
76            'folio_number': str(folio),
77            'side': side,
78            'page_type': 'PAGE',
79            'sort_order': sort_order,
80        }
81
82    # Cover and guard pages
83    if 'antcop' in filename or 'postcop' in filename:
84        sort_order = -2 if 'antcop' in filename else 99999
85        return {
86            'folio_number': None,
87            'side': None,
88            'page_type': 'COVER',
89            'sort_order': sort_order,
90        }
91
92    if 'fdg' in filename or 'risg' in filename:
93        sort_order = -1
94        return {
95            'folio_number': None,
96            'side': None,
97            'page_type': 'GUARD',
98            'sort_order': sort_order,
99        }
100
101    # Marginalia detail photos
102    m = re.match(r'Siena\s*[Hh][Pp]\s*(\d+)\.jpg', filename, re.IGNORECASE)
103    if m:
104        num = int(m.group(1))
105        return {
106            'folio_number': None,
107            'side': None,
108            'page_type': 'MARGINALIA_DETAIL',
109            'sort_order': 100000 + num,
110        }
111
112    return {
113        'folio_number': None,
114        'side': None,
115        'page_type': 'OTHER',
116        'sort_order': 999999,
117    }

120def catalog_manuscript(conn, shelfmark, parser_fn):
121    """Catalog all images for a manuscript."""
122    cur = conn.cursor()
123    cur.execute("SELECT id, image_dir FROM manuscripts WHERE shelfmark = ?", (shelfmark,))
124    row = cur.fetchone()
125    if not row:
126        print(f"  WARNING: Manuscript {shelfmark} not found in database")
127        return 0
128
129    ms_id, image_dir = row
130    img_path = BASE_DIR / image_dir
131    if not img_path.exists():
132        print(f"  WARNING: Image directory not found: {img_path}")
133        return 0
134
135    count = 0
136    for f in sorted(img_path.glob('*.jpg')):
137        parsed = parser_fn(f.name)
138        relative = f"{image_dir}/{f.name}"
139        cur.execute(
140            """INSERT OR IGNORE INTO images
141               (manuscript_id, filename, folio_number, side, page_type, sort_order, relative_path)
142               VALUES (?, ?, ?, ?, ?, ?, ?)""",
143            (ms_id, f.name, parsed['folio_number'], parsed['side'],
144             parsed['page_type'], parsed['sort_order'], relative)
145        )
146        count += 1
147
148    conn.commit()
149    return count

152def main():
153    conn = sqlite3.connect(DB_PATH)
154
155    print("Cataloging BL C.60.o.12 images...")
156    bl_count = catalog_manuscript(conn, 'C.60.o.12', parse_bl_filename)
157    print(f"  {bl_count} images cataloged")
158
159    print("Cataloging Siena O.III.38 images...")
160    siena_count = catalog_manuscript(conn, 'O.III.38', parse_siena_filename)
161    print(f"  {siena_count} images cataloged")
162
163    # Summary stats
164    cur = conn.cursor()
165    for page_type in ['PAGE', 'MARGINALIA_DETAIL', 'COVER', 'GUARD', 'OTHER']:
166        cur.execute("SELECT COUNT(*) FROM images WHERE page_type = ?", (page_type,))
167        print(f"  {page_type}: {cur.fetchone()[0]}")
168
169    conn.close()
170    print("Done.")
