382def main():
383    conn = sqlite3.connect(DB_PATH)
384    cur = conn.cursor()
385
386    # Check which terms are VERIFIED (do not touch)
387    cur.execute("SELECT slug, review_status FROM dictionary_terms")
388    statuses = {row[0]: row[1] for row in cur.fetchall()}
389
390    # Build staging artifact
391    staging_data = []
392    updated = 0
393    skipped = 0
394
395    for slug, sig in SIGNIFICANCE.items():
396        status = statuses.get(slug)
397        if status == 'VERIFIED':
398            print(f"  SKIP (VERIFIED): {slug}")
399            skipped += 1
400            continue
401
402        if slug not in statuses:
403            print(f"  SKIP (not in DB): {slug}")
404            continue
405
406        hp_text = sig.get('hp', '')
407        schol_text = sig.get('scholarship', '')
408
409        staging_data.append({
410            'slug': slug,
411            'significance_to_hp': hp_text,
412            'significance_to_scholarship': schol_text,
413            'source_method': 'LLM_ASSISTED',
414            'review_status': 'DRAFT',
415            'confidence': 'MEDIUM',
416        })
417
418        cur.execute("""
419            UPDATE dictionary_terms
420            SET significance_to_hp = ?,
421                significance_to_scholarship = ?,
422                source_method = COALESCE(source_method, 'LLM_ASSISTED'),
423                confidence = COALESCE(confidence, 'MEDIUM'),
424                updated_at = ?
425            WHERE slug = ?
426              AND (review_status IS NULL OR review_status != 'VERIFIED')
427        """, (hp_text, schol_text, datetime.now().isoformat(), slug))
428
429        if cur.rowcount > 0:
430            updated += 1
431            print(f"  UPDATED: {slug}")
432
433    conn.commit()
434    conn.close()
435
436    # Write staging artifact
437    staging_path = STAGING_DIR / "dictionary_significance.json"
438    with open(staging_path, 'w', encoding='utf-8') as f:
439        json.dump(staging_data, f, indent=2, ensure_ascii=False)
440
441    print(f"\nUpdated {updated} terms, skipped {skipped} verified.")
442    print(f"Staging artifact: {staging_path}")
443
444
445if __name__ == "__main__":
446    print("=== Generating Dictionary Significance Prose ===\n")
447    main()
